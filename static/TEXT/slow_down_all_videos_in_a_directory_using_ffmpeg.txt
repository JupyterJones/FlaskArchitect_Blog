'''
This will slow all videos in a directory using ffmpeg,
how do i slow all videos in a directory, using ffmpeg and python        
'''
import glob
import subprocess
def slowem(video_list):
    cnt = 0
    for vid in vid_list:
        cnt += 1
        cmd = ["ffmpeg", "-i", vid, "-filter:v", "setpts=2*PTS", "-c:a", "copy", "-y", f"cnt_{cnt}.mp4"]
        subprocess.run(cmd)
    return print('All Slowed')
if __name__=='main':    
    vid_list = glob.glob("*.mp4")
    slowem(video_list)
