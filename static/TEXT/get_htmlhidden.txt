#!/home/jack/miniconda3/envs/cloned_base/bin/python
import os
def main():
    HTML_CONTENT=[]
    directory=os.path.join(os.getcwd(), 'templates/')
    dir_name = directory.replace('/','_')+'.txt'
    print('/home/jack/hidden/'+dir_name)
    all_ = open('/home/jack/'+dir_name,'w')
    htmls = [os.path.join(directory, f) for f in os.listdir(directory) if f.endswith('.html')]
    for filename in htmls:
        file_name = filename.replace('/','_')
        print(filename)
        print(file_name)
        contents=open(filename).read()
        all_.write('\n-----location-----\n') 
        all_.write('<!-- '+file_name+' -->\n')
        all_.write(contents)
        HTML_CONTENT.append([contents])
        all_.write('\n')
        all_.close()
    return HTML_CONTENT
if __name__ == '__main__':    
    main()
